рЃњрЃљрЃЏрЃћрЃЮрЃарЃћрЃЉрЃљ рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў рЃбрЃўрЃърЃўрЃА рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃў рЃАрЃљрЃ«рЃџрЃўрЃА рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃљ

­Ъџђ JavaScript рЃЏрЃћрЃюрЃбрЃЮрЃарЃБрЃџрЃў рЃАрЃћрЃАрЃўрЃљ

JavaScript-рЃўрЃА рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃЎрЃЮрЃюрЃфрЃћрЃцрЃфрЃўрЃћрЃЉрЃў рЃЊрЃљ рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃћрЃЉрЃў

­ЪЊџ рЃгрЃўрЃюрЃљ рЃЏрЃљрЃАрЃљрЃџрЃўрЃА рЃњрЃљрЃЏрЃћрЃЮрЃарЃћрЃЉрЃљ

JavaScript рЃАрЃўрЃюрЃбрЃљрЃЦрЃАрЃў

JavaScript-рЃўрЃА рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃАрЃўрЃюрЃбрЃљрЃЦрЃАрЃўрЃА рЃгрЃћрЃАрЃћрЃЉрЃў:

// рЃЎрЃЮрЃЏрЃћрЃюрЃбрЃљрЃарЃћрЃЉрЃў /* рЃЏрЃарЃљрЃЋрЃљрЃџрЃ«рЃљрЃќрЃўрЃљрЃюрЃў рЃЎрЃЮрЃЏрЃћрЃюрЃбрЃљрЃарЃў */ // рЃгрЃћрЃарЃбрЃўрЃџрЃЏрЃФрЃўрЃЏрЃћ рЃљрЃарЃљ рЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃДрЃЮрЃЋрЃћрЃџ рЃ«рЃљрЃќрЃўрЃА рЃЉрЃЮрЃџрЃЮрЃА console.log("рЃњрЃљрЃЏрЃљрЃарЃ»рЃЮрЃЉрЃљ"); // Case-sensitive рЃћрЃюрЃљрЃљ let myVariable = "test"; let MyVariable = "рЃАрЃ«рЃЋрЃљ рЃбрЃћрЃАрЃбрЃў"; // рЃћрЃА рЃАрЃ«рЃЋрЃљ рЃфрЃЋрЃџрЃљрЃЊрЃўрЃљ

рЃфрЃЋрЃџрЃљрЃЊрЃћрЃЉрЃў

рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃў: let рЃЊрЃљ const рЃўрЃДрЃћрЃюрЃћрЃЉрЃљ var-рЃўрЃА рЃюрЃљрЃфрЃЋрЃџрЃљрЃЊ рЃЌрЃљрЃюрЃљрЃЏрЃћрЃЊрЃарЃЮрЃЋрЃћ JavaScript-рЃерЃў
// let - рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃерЃћрЃфрЃЋрЃџрЃљ let age = 25; age = 26; // рЃЏрЃБрЃерЃљрЃЮрЃЉрЃА // const - рЃЋрЃћрЃа рЃерЃћрЃўрЃфрЃЋрЃџрЃћрЃЉрЃљ const name = "John"; // name = "рЃњрЃўрЃЮрЃарЃњрЃў"; // рЃерЃћрЃфрЃЊрЃЮрЃЏрЃљрЃА рЃњрЃљрЃЏрЃЮрЃўрЃгрЃЋрЃћрЃЋрЃА // var - рЃФрЃЋрЃћрЃџрЃў рЃЏрЃћрЃЌрЃЮрЃЊрЃў (рЃљрЃа рЃњрЃўрЃарЃЕрЃћрЃЋрЃЌ) var oldWay = "рЃФрЃЋрЃћрЃџрЃў рЃАрЃўрЃюрЃбрЃљрЃЦрЃАрЃў";

рЃўрЃюрЃбрЃћрЃарЃљрЃЦрЃбрЃўрЃБрЃџрЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:

NPM рЃЊрЃљ JavaScript рЃњрЃљрЃарЃћрЃЏрЃЮ

Node Package Manager (NPM) JavaScript-рЃўрЃА рЃърЃљрЃЎрЃћрЃбрЃћрЃЉрЃўрЃА рЃЏрЃћрЃюрЃћрЃ»рЃћрЃарЃўрЃљ:

// package.json рЃцрЃљрЃўрЃџрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ npm init -y // рЃърЃљрЃЎрЃћрЃбрЃўрЃА рЃўрЃюрЃАрЃбрЃљрЃџрЃљрЃфрЃўрЃљ npm install lodash // рЃърЃљрЃЎрЃћрЃбрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ const _ = require('lodash');

рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃбрЃўрЃърЃћрЃЉрЃў

рЃбрЃўрЃърЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў рЃљрЃдрЃгрЃћрЃарЃљ
Number 42, 3.14 рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃў
String "рЃбрЃћрЃЦрЃАрЃбрЃў", 'рЃбрЃћрЃЦрЃАрЃбрЃў' рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў
Boolean true, false рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ
Undefined undefined рЃњрЃљрЃюрЃБрЃАрЃљрЃќрЃдрЃЋрЃарЃћрЃџрЃў
Null null рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ
Object {key: "value"} рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў
Array [1, 2, 3] рЃЏрЃљрЃАрЃўрЃЋрЃў

­Ъћё рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃўрЃА рЃЊрЃљ рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў

рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў

// рЃЌрЃљрЃюрЃљрЃАрЃгрЃЮрЃарЃЮрЃЉрЃўрЃА рЃерЃћрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃљ == // рЃбрЃўрЃърЃўрЃА рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃўрЃЌ === // рЃќрЃБрЃАрЃбрЃў рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃљ (рЃбрЃўрЃърЃўрЃЌрЃљ рЃЊрЃљ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃўрЃЌ) // рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃћрЃЉрЃў 5 == "5" // true (рЃњрЃљрЃарЃЊрЃљрЃўрЃЦрЃЏрЃюрЃћрЃЉрЃљ) 5 === "5" // false (рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃбрЃўрЃърЃўрЃљ) // рЃБрЃЌрЃљрЃюрЃљрЃАрЃгрЃЮрЃарЃЮрЃЉрЃљ != // рЃбрЃўрЃърЃўрЃА рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃўрЃЌ !== // рЃќрЃБрЃАрЃбрЃў рЃБрЃЌрЃљрЃюрЃљрЃАрЃгрЃЮрЃарЃЮрЃЉрЃљ // рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃљ > <>= <=

рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃўрЃА рЃбрЃћрЃАрЃбрЃў:

рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў

// AND (рЃЊрЃљ) && // рЃЮрЃарЃўрЃЋрЃћ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЮрЃА true // OR (рЃљрЃю) || // рЃћрЃарЃЌ-рЃћрЃарЃЌрЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЮрЃА true // NOT (рЃљрЃарЃљ) ! // рЃдрЃўрЃарЃћрЃЉрЃБрЃџрЃћрЃЉрЃўрЃА рЃерЃћрЃЉрЃарЃБрЃюрЃћрЃЉрЃљ // рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃћрЃЉрЃў true && false // false true || false // true !true // false

If/Else рЃњрЃљрЃюрЃфрЃ«рЃљрЃЊрЃћрЃЉрЃћрЃЉрЃў

let score = 85; if (score >= 90) { console.log("рЃерЃћрЃАрЃљрЃюрЃўрЃерЃюрЃљрЃЋрЃў!"); } else if (score >= 70) { console.log("рЃЎрЃљрЃарЃњрЃў!"); } else if (score >= 60) { console.log("рЃАрЃљрЃерЃБрЃљрЃџрЃЮ"); } else { console.log("рЃњрЃљрЃБрЃЏрЃ»рЃЮрЃЉрЃћрЃАрЃћрЃЉрЃўрЃА рЃАрЃљрЃГрЃўрЃарЃЮрЃћрЃЉрЃљрЃљ"); }

рЃЦрЃБрЃџрЃўрЃА рЃерЃћрЃцрЃљрЃАрЃћрЃЉрЃљ:

Switch/Case

let day = 3; let dayName; switch (day) { case 1: dayName = "рЃЮрЃарЃерЃљрЃЉрЃљрЃЌрЃў"; break; case 2: dayName = "рЃАрЃљрЃЏрЃерЃљрЃЉрЃљрЃЌрЃў"; break; case 3: dayName = "рЃЮрЃЌрЃ«рЃерЃљрЃЉрЃљрЃЌрЃў"; break; case 4: dayName = "рЃ«рЃБрЃЌрЃерЃљрЃЉрЃљрЃЌрЃў"; break; case 5: dayName = "рЃърЃљрЃарЃљрЃАрЃЎрЃћрЃЋрЃў"; break; default: dayName = "рЃерЃљрЃЉрЃљрЃЌ-рЃЎрЃЋрЃўрЃарЃљ"; }

Ternary рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃў

// рЃАрЃўрЃюрЃбрЃљрЃЦрЃАрЃў: condition ? value_if_true : value_if_false let age = 20; let status = age >= 18 ? "рЃќрЃарЃЊрЃљрЃАрЃарЃБрЃџрЃў" : "рЃљрЃарЃљрЃАрЃарЃБрЃџрЃгрЃџрЃЮрЃЋрЃљрЃюрЃў"; // рЃарЃЌрЃБрЃџрЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў let grade = 85; let result = grade >= 90 ? "A" : grade >= 80 ? "B" : grade >= 70 ? "C" : grade >= 60 ? "D" : "F";

рЃљрЃАрЃљрЃЎрЃўрЃА рЃерЃћрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃљ:

­Ъћё рЃбрЃўрЃърЃўрЃА рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ (Type Coercion)

рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃўрЃА рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ

// String() рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃЌ String(123) // "123" String(true) // "true" String(null) // "null" // toString() рЃЏрЃћрЃЌрЃЮрЃЊрЃўрЃЌ let num = 42; num.toString() // "42" // + рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃўрЃЌ (concatenation) 123 + "" // "123" true + "" // "true"

рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃљрЃЊ рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ:

рЃарЃўрЃфрЃ«рЃЋрЃўрЃА рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ

// Number() рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃЌ Number("123") // 123 Number("12.5") // 12.5 Number(true) // 1 Number(false) // 0 Number("") // 0 Number("hello") // NaN // parseInt() рЃЊрЃљ parseFloat() parseInt("123px") // 123 parseFloat("12.5kg") // 12.5 // + рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃўрЃЌ (рЃБрЃюрЃљрЃарЃБрЃџ рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ) +"123" // 123 +true // 1
рЃДрЃБрЃарЃљрЃЊрЃдрЃћрЃЉрЃљ: NaN (Not a Number) рЃюрЃўрЃерЃюрЃљрЃЋрЃА, рЃарЃЮрЃЏ рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ рЃЋрЃћрЃа рЃЏрЃЮрЃ«рЃћрЃарЃ«рЃЊрЃљ

Boolean рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ

// Boolean() рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃЌ Boolean(1) // true Boolean(0) // false Boolean("") // false Boolean("hello") // true Boolean(null) // false Boolean(undefined) // false // !! рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃўрЃЌ !!1 // true !!0 // false !!"hello" // true !!"" // false
Falsy рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃў: false, 0, "", null, undefined, NaN
рЃДрЃЋрЃћрЃџрЃљ рЃАрЃ«рЃЋрЃљ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ рЃљрЃарЃўрЃА Truthy

Boolean рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃўрЃА рЃбрЃћрЃАрЃбрЃў:

рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ

// рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃў рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃљрЃЊ let obj = {name: "John", age: 25}; String(obj) // "[object Object]" JSON.stringify(obj) // '{"name":"John","age":25}' // рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃў рЃарЃўрЃфрЃ«рЃЋрЃљрЃЊ Number(obj) // NaN Number({}) // NaN Number([]) // 0 Number([1]) // 1 Number([1, 2]) // NaN

­ЪЌѓ№ИЈ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃў рЃЊрЃљ рЃфрЃўрЃЎрЃџрЃћрЃЉрЃў

рЃЏрЃљрЃАрЃўрЃЋрЃћрЃЉрЃў (Arrays)

// рЃЏрЃљрЃАрЃўрЃЋрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ let fruits = ["рЃЋрЃљрЃерЃџрЃў", "рЃЉрЃљрЃюрЃљрЃюрЃў", "рЃДрЃЋрЃўрЃЌрЃћрЃџрЃў"]; let numbers = [1, 2, 3, 4, 5]; let mixed = ["рЃбрЃћрЃЦрЃАрЃбрЃў", 42, true, null]; // рЃЏрЃљрЃАрЃўрЃЋрЃўрЃА рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃќрЃћ рЃгрЃЋрЃЊрЃЮрЃЏрЃљ console.log(fruits[0]); // "рЃЋрЃљрЃерЃџрЃў" console.log(fruits.length); // 3 // рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ fruits.push("рЃДрЃБрЃарЃФрЃћрЃюрЃў"); // рЃЉрЃЮрЃџрЃЮрЃерЃў рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ fruits.unshift("рЃљрЃбрЃљрЃЏрЃў"); // рЃЊрЃљрЃАрЃљрЃгрЃДрЃўрЃАрЃерЃў рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ // рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃгрЃљрЃерЃџрЃљ fruits.pop(); // рЃЉрЃЮрЃџрЃЮ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃгрЃљрЃерЃџрЃљ fruits.shift(); // рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃгрЃљрЃерЃџрЃљ

рЃЏрЃљрЃАрЃўрЃЋрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃћрЃЉрЃў:

рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃў (Objects)

// рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ let person = { name: "John", age: 25, city: "рЃЌрЃЉрЃўрЃџрЃўрЃАрЃў", isStudent: false }; // рЃЌрЃЋрЃўрЃАрЃћрЃЉрЃћрЃЉрЃќрЃћ рЃгрЃЋрЃЊрЃЮрЃЏрЃљ - Dot notation console.log(person.name); // "John" console.log(person.age); // 25 // рЃЌрЃЋрЃўрЃАрЃћрЃЉрЃћрЃЉрЃќрЃћ рЃгрЃЋрЃЊрЃЮрЃЏрЃљ - Bracket notation console.log(person["name"]); // "John" console.log(person["age"]); // 25 // рЃљрЃ«рЃљрЃџрЃў рЃЌрЃЋрЃўрЃАрЃћрЃЉрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ person.profession = "рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃўрЃАрЃбрЃў"; person["email"] = "nika@example.com";

Dot vs Bracket Notation

let car = { brand: "Toyota", model: "Camry", year: 2020, "engine-type": "V6" // рЃЊрЃћрЃцрЃўрЃАрЃўрЃљрЃюрЃў рЃЎрЃў рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃБрЃџрЃў рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљрЃљ }; // Dot notation - рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃЊрЃљ рЃАрЃгрЃарЃљрЃцрЃў console.log(car.brand); // "Toyota" // Bracket notation - рЃЏрЃћрЃбрЃў рЃЏрЃЮрЃЦрЃюрЃўрЃџрЃЮрЃЉрЃљ console.log(car["brand"]); // "Toyota" console.log(car["engine-type"]); // "V6" - dot notation рЃЋрЃћрЃа рЃўрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃА // рЃЊрЃўрЃюрЃљрЃЏрЃўрЃБрЃарЃў рЃгрЃЋрЃЊрЃЮрЃЏрЃљ let property = "model"; console.log(car[property]); // "Camry"
рЃарЃљ рЃБрЃюрЃЊрЃљ рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ?
Рђб Dot notation - рЃЌрЃБ рЃЋрЃўрЃфрЃўрЃЌ рЃЌрЃЋрЃўрЃАрЃћрЃЉрЃўрЃА рЃАрЃљрЃ«рЃћрЃџрЃў
Рђб Bracket notation - рЃЊрЃўрЃюрЃљрЃЏрЃўрЃБрЃарЃў рЃгрЃЋрЃЊрЃЮрЃЏрЃўрЃАрЃЌрЃЋрЃўрЃА рЃљрЃю рЃБрЃфрЃюрЃљрЃБрЃарЃў рЃАрЃљрЃ«рЃћрЃџрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА

рЃфрЃўрЃЎрЃџрЃћрЃЉрЃў (Loops)

// For loop for (let i = 0; i < 5; i++) { console.log("рЃфрЃўрЃЎрЃџрЃўрЃА рЃЏрЃћрЃЮрЃарЃћ:", i); } // While loop let count=0; while (count < 3) { console.log("Count:", count); count++; } // For...of (рЃЏрЃљрЃАрЃўрЃЋрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА) let colors=["рЃгрЃўрЃЌрЃћрЃџрЃў", "рЃЏрЃгрЃЋрЃљрЃюрЃћ" , "рЃџрЃБрЃарЃ»рЃў" ]; for (let color of colors) { console.log(color); } // For...in (рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА) let student={name: "рЃљрЃюрЃљ" , grade: 95, subject: "рЃЏрЃљрЃЌрЃћрЃЏрЃљрЃбрЃўрЃЎрЃљ" }; for (let key in student) { console.log(key + ": " + student[key]); }

рЃфрЃўрЃЎрЃџрЃћрЃЉрЃўрЃА рЃЊрЃћрЃЏрЃЮрЃюрЃАрЃбрЃарЃљрЃфрЃўрЃљ:

рЃЏрЃљрЃАрЃўрЃЋрЃўрЃА рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў

let numbers = [1, 2, 3, 4, 5]; // forEach - рЃДрЃЋрЃћрЃџрЃљ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃќрЃћ рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃљ numbers.forEach(num => console.log(num * 2)); // map - рЃљрЃ«рЃљрЃџрЃў рЃЏрЃљрЃАрЃўрЃЋрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃўрЃЌ let doubled = numbers.map(num => num * 2); console.log(doubled); // [2, 4, 6, 8, 10] // filter - рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃЏрЃўрЃ«рЃћрЃЊрЃЋрЃўрЃЌ рЃцрЃўрЃџрЃбрЃарЃљрЃфрЃўрЃљ let evens = numbers.filter(num => num % 2 === 0); console.log(evens); // [2, 4] // find - рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃў рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃърЃўрЃарЃЮрЃЉрЃљрЃА рЃљрЃЎрЃЏрЃљрЃДрЃЮрЃцрЃўрЃџрЃћрЃЉрЃА let found = numbers.find(num => num > 3); console.log(found); // 4 // reduce - рЃћрЃарЃЌ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљрЃЊ рЃЊрЃљрЃДрЃЋрЃљрЃюрЃљ let sum = numbers.reduce((acc, num) => acc + num, 0); console.log(sum); // 15

рЃЏрЃљрЃАрЃўрЃЋрЃўрЃА рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃўрЃА рЃЊрЃћрЃЏрЃЮрЃюрЃАрЃбрЃарЃљрЃфрЃўрЃљ:

­ЪЊЮ рЃАрЃљрЃ«рЃџрЃўрЃА рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃљ

рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃљ 1: рЃЦрЃБрЃџрЃћрЃЉрЃўрЃА рЃЎрЃљрЃџрЃЎрЃБрЃџрЃљрЃбрЃЮрЃарЃў

рЃерЃћрЃЦрЃЏрЃћрЃюрЃўрЃЌ рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃЏрЃўрЃўрЃдрЃћрЃЉрЃА рЃЦрЃБрЃџрЃћрЃЉрЃўрЃА рЃЏрЃљрЃАрЃўрЃЋрЃА рЃЊрЃљ рЃЊрЃљрЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА:

  • рЃАрЃљрЃерЃБрЃљрЃџрЃЮ рЃЦрЃБрЃџрЃљ
  • рЃЏрЃљрЃЦрЃАрЃўрЃЏрЃљрЃџрЃБрЃарЃў рЃЦрЃБрЃџрЃљ
  • рЃЏрЃўрЃюрЃўрЃЏрЃљрЃџрЃБрЃарЃў рЃЦрЃБрЃџрЃљ
  • рЃЦрЃБрЃџрЃћрЃЉрЃўрЃА рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљ 90-рЃќрЃћ рЃЏрЃћрЃбрЃў
function analyzeScores(scores) { // рЃЌрЃЦрЃЋрЃћрЃюрЃў рЃЎрЃЮрЃЊрЃў рЃљрЃЦ } // рЃбрЃћрЃАрЃбрЃў let testScores = [85, 92, 78, 96, 88, 91, 84]; console.log(analyzeScores(testScores));

рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃљ 2: рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃЏрЃћрЃюрЃћрЃ»рЃЏрЃћрЃюрЃбрЃў

рЃерЃћрЃЦрЃЏрЃћрЃюрЃўрЃЌ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃўрЃюрЃцрЃЮрЃарЃЏрЃљрЃфрЃўрЃўрЃА рЃерЃћрЃАрЃљрЃюрЃљрЃ«рЃљрЃЊ:

let classroom = { students: [], addStudent: function(name, age, grade) { // рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃўрЃА рЃџрЃЮрЃњрЃўрЃЎрЃљ }, findStudent: function(name) { // рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃўрЃА рЃЏрЃЮрЃФрЃћрЃЉрЃюрЃўрЃА рЃџрЃЮрЃњрЃўрЃЎрЃљ }, getAverageAge: function() { // рЃАрЃљрЃерЃБрЃљрЃџрЃЮ рЃљрЃАрЃљрЃЎрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃљ }, getTopStudents: function(minGrade) { // рЃЏрЃљрЃдрЃљрЃџрЃў рЃЦрЃБрЃџрЃћрЃЉрЃўрЃА рЃЏрЃЦрЃЮрЃюрЃћ рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃћрЃЉрЃў } };

рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃљ 3: Type Coercion рЃњрЃљрЃЏрЃЮрЃгрЃЋрЃћрЃЋрЃћрЃЉрЃў

рЃњрЃљрЃюрЃЏрЃљрЃарЃбрЃћрЃЌ рЃарЃљ рЃўрЃЦрЃюрЃћрЃЉрЃљ рЃерЃћрЃЊрЃћрЃњрЃў рЃДрЃЋрЃћрЃџрЃљ рЃ«рЃљрЃќрЃўрЃАрЃЌрЃЋрЃўрЃА:

console.log(1 + "2" + "2"); // ? console.log(1 + +"2" + "2"); // ? console.log(1 + -"1" + "2"); // ? console.log(+"1" + "1" + "2"); // ? console.log("A" - "B" + "2"); // ? console.log("A" - "B" + 2); // ? console.log("0" == false); // ? console.log("0" === false); // ? console.log(null == undefined); // ? console.log(null === undefined); // ?

рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃљ 4: рЃерЃЮрЃърЃўрЃюрЃњ рЃЎрЃљрЃџрЃљрЃЌрЃљ

рЃерЃћрЃЦрЃЏрЃћрЃюрЃўрЃЌ рЃерЃЮрЃърЃўрЃюрЃњ рЃЎрЃљрЃџрЃљрЃЌрЃўрЃА рЃАрЃўрЃАрЃбрЃћрЃЏрЃљ рЃарЃЮрЃЏрЃћрЃџрЃАрЃљрЃф рЃерЃћрЃБрЃФрЃџрЃўрЃљ:

  • рЃърЃарЃЮрЃЊрЃБрЃЦрЃбрЃћрЃЉрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ (рЃАрЃљрЃ«рЃћрЃџрЃў, рЃцрЃљрЃАрЃў, рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљ)
  • рЃърЃарЃЮрЃЊрЃБрЃЦрЃбрЃћрЃЉрЃўрЃА рЃгрЃљрЃерЃџрЃљ
  • рЃАрЃљрЃћрЃарЃЌрЃЮ рЃдрЃўрЃарЃћрЃЉрЃБрЃџрЃћрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃљ
  • рЃДрЃЋрЃћрЃџрЃљрЃќрЃћ рЃФрЃЋрЃўрЃарЃў рЃърЃарЃЮрЃЊрЃБрЃЦрЃбрЃўрЃА рЃЏрЃЮрЃФрЃћрЃЉрЃюрЃљ
  • рЃърЃарЃЮрЃЊрЃБрЃЦрЃбрЃћрЃЉрЃўрЃА рЃцрЃўрЃџрЃбрЃарЃљрЃфрЃўрЃљ рЃцрЃљрЃАрЃўрЃА рЃЏрЃўрЃ«рЃћрЃЊрЃЋрЃўрЃЌ
